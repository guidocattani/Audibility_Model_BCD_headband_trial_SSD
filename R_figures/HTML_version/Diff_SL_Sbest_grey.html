read_diff_SL <- function(path_in, sht) {
  dt <- read_excel(path_in, sheet = sht)
  dt <- dt[ -c(1, 4, 5, 6, 7, 8, 9, 21, 22)]
  # rename columns
  clm = colnames(dt)
  clm= clm[-c(1, 2)]
  newco <- c()
  for (fb in clm){
      f <- strsplit(fb, "_")[[1]][1]
      newco <- c(newco, f)
  }   

newco <- c('Study_ID', 'Device', newco)
print(newco)
colnames(dt) <- newco
  return(dt)
}
